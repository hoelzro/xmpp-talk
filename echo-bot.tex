    my ( undef, undef, $contact ) = @_;

    $contact->send_subscribed;
});
\end{minted}

$conn->reg_cb(message => sub {
    my ( undef, $msg ) = @_;

    my $reply = $msg->make_reply;
    $reply->add_body($msg->body);
    $reply->send;
});
\end{minted}
