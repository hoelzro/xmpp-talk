\newpage
\begin{shaded}
\begin{minted}{perl}
my $conn = AnyEvent::XMPP::IM::Connection->new(
    username => 'bot',
    domain   => 'chat.madmongers',
    resource => scalar(getpwuid $<),
    host     => 'localhost',
    port     => 5222,
    password => 'abc123',
);
\end{minted}
\end{shaded}

\begin{shaded}
\begin{minted}{perl}
$conn->reg_cb(session_ready => sub {
    my $msg = AnyEvent::XMPP::IM::Message->new(
        to   => $to,
        body => 'Hello, World!',
    );

    $msg->send($conn);

    my $timer;
    $timer = AnyEvent->timer(
        after => 3,
        cb    => sub {
            undef $timer;
            $cond->send;
        },
    );
});
\end{minted}
\end{shaded}

\begin{shaded}
\begin{minted}{perl}
$conn->connect;
\end{minted}
\end{shaded}
